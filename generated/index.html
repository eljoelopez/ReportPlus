def get_data_frame(base:Base):
    id = []
    name = []
    volume = []
    area = []

    for e in base.elements:
        id.append(e.id)
        name.append(e.speckleType.split(".")[-1])
        volume.append(e.volume)
        area.append(e.area)

    data = {
        "id" : id,
        "Name" : name,
        "Volume" : volume,
        "Area" : area
    }

    return pd.DataFrame(data=data)
